function countByFilter(id) {
  return id === 'all' ? ARTICLES.length : ARTICLES.filter(a => a.cat === id).length;
}

function render() {
  const q = document.getElementById('search').value.toLowerCase().trim();

  document.getElementById('filter-tabs').innerHTML = CATS.map(c => {
    const cnt = countByFilter(c.id);
    if (!cnt && c.id !== 'all') return '';
    return `
${c.label} ${cnt}
`;
  }).join('');

  const filtered = ARTICLES.filter(a => {
    const matchCat = activeFilter === 'all' || a.cat === activeFilter;
    const matchQ = !q || a.title.toLowerCase().includes(q) || a.desc.toLowerCase().includes(q);
    return matchCat && matchQ;
  });

  const byCat = CATS.filter(c => c.id !== 'all').map(c => ({...c, cnt: ARTICLES.filter(a => a.cat === c.id).length}));
  document.getElementById('stats-row').innerHTML =
    `
${ARTICLES.length} Total Articles
` +
    byCat.map(c => `
${c.cnt} ${c.label}
`).join('');

  document.getElementById('results-count').innerHTML =
    `Showing ${filtered.length} of ${ARTICLES.length} articles${q ? ` matching "${q}"` : ''}`;

  if (!filtered.length) {
    document.getElementById('output').innerHTML = '
No articles match your search.
';
    return;
  }

  const grouped = {};
  filtered.forEach(a => { (grouped[a.cat] = grouped[a.cat] || []).push(a); });
  const catOrder = CATS.filter(c => c.id !== 'all').map(c => c.id);

  document.getElementById('output').innerHTML = catOrder.filter(id => grouped[id]).map(id => {
    const cat = CATS.find(c => c.id === id);
    return `
      

        

          

          
${cat.label}

          ${grouped[id].length}
        

        

          ${grouped[id].map(a => {
            const idx = ARTICLES.indexOf(a);
            const pl = a.priority === 'high' ? 'High Priority' : a.priority === 'med' ? 'Medium Priority' : 'Lower Priority';
            return `
            

              
${a.title}

              
${a.desc}

              

                
                
              

              

                ${pl}
                
              

            
`;
          }).join('')}
        

      
`;
  }).join('');
}

function setFilter(id) { activeFilter = id; render(); }

async function openArticle(idx) {
  const apiKey = getApiKey();
  if (!apiKey || !apiKey.startsWith('sk-ant-')) {
    alert('Please paste your Anthropic API key at the top of the page before generating an article.');
    return;
  }

  currentArticle = ARTICLES[idx];
  generatedData = null;
  currentView = 'preview';

  // Grab image URL from the card input if provided
  const imgInput = document.getElementById('img-' + idx);
  currentImageUrl = imgInput ? imgInput.value.trim() : '';

  document.getElementById('library-view').style.display = 'none';
  document.getElementById('article-view').style.display = 'block';
  document.getElementById('topbar-title').textContent = currentArticle.title;
  document.getElementById('btn-preview').classList.add('active');
  document.getElementById('btn-code').classList.remove('active');

  showState('loading');
  await generateArticle();
}

function closeArticle() {
  document.getElementById('article-view').style.display = 'none';
  document.getElementById('library-view').style.display = 'block';
  window.scrollTo(0, 0);
}

async function retryGenerate() {
  showState('loading');
  await generateArticle();
}

function getApiKey() {
  return document.getElementById('api-key-input').value.trim();
}

function checkApiKey() {
  const key = getApiKey();
  const status = document.getElementById('api-key-status');
  if (!key) { status.textContent = ''; status.className = ''; return; }
  if (key.startsWith('sk-ant-')) {
    status.textContent = 'Key looks good';
    status.className = 'valid';
  } else {
    status.textContent = 'Key should start with sk-ant-';
    status.className = 'invalid';
  }
}

function toggleKeyVisibility() {
  const input = document.getElementById('api-key-input');
  const btn = document.getElementById('toggle-btn');
  if (input.type === 'password') { input.type = 'text'; btn.textContent = 'Hide'; }
  else { input.type = 'password'; btn.textContent = 'Show'; }
}

You write SEO-optimized content for their website targeting engineers, procurement managers, and purchasing specialists across aerospace, defense, food processing, pharmaceutical, marine, and manufacturing industries. Tone: professional, authoritative, technically accurate, and accessible. No emojis anywhere in the output. Always refer to Plastic-Craft in the third person ΓÇË "Plastic-Craft offers..." or "the team at Plastic-Craft..." ΓÇË never first person plural.`,
        messages: [{

Return ONLY a valid JSON object. No markdown, no code fences, no extra text before or after. Structure:

  "metaDesc": "150-160 character SEO meta description. End with a period.",
  "intro": "2-3 sentence engaging opening paragraph that hooks the reader and establishes the topic's relevance.",
  "sections": [
    {
      "h2": "Section heading ΓÇË specific and compelling, no colons",
      "body": "2-3 paragraphs of content separated by double newlines. Use 
  • text
 HTML list syntax where a list genuinely helps. Do not use bullet characters ΓÇË only proper HTML list tags."
    }
  ],
  "cta_heading": "Short action-oriented heading, 8 words max",
  "cta_body": "2-3 sentences inviting the reader to contact Plastic-Craft. Naturally mention West Nyack NY and either the phone (845) 358-3010 or email [email protected]."
}

Requirements: 4-5 sections. 700-900 total words of body content. No emojis. No bullet characters. Each section body is 2-3 paragraphs. Mention Plastic-Craft's certifications (ISO 9001:2015 and/or AS9100D) naturally in at least one section where it is relevant. Write in third person about Plastic-Craft.`

      })
    });

    const data = await response.json();
    if (!data.content) throw new Error('No content returned');

    const rawText = data.content.filter(b => b.type === 'text').map(b => b.text).join('');
    const cleaned = rawText.replace(/```json\n?/g, '').replace(/```\n?/g, '').trim();
    generatedData = JSON.parse(cleaned);

    renderPreview();
    renderCode();
    showState('preview');
    switchView('preview');

  } catch (err) {
    console.error('Generation error:', err);
    showState('error');
  }

function showState(state) {
  ['loading', 'error', 'preview', 'code'].forEach(s => {
    document.getElementById('state-' + s).style.display = 'none';
  });
  document.getElementById('state-' + state).style.display = 'block';
}

function switchView(view) {
  currentView = view;
  document.getElementById('btn-preview').classList.toggle('active', view === 'preview');
  document.getElementById('btn-code').classList.toggle('active', view === 'code');
  if (!generatedData) return;
  showState(view);
}

function renderPreview() {
  if (!generatedData) return;
  const d = generatedData;
  const title = currentArticle.title;

  document.getElementById('preview-meta').textContent = d.metaDesc;

  // Build image block ΓÇË real image if URL provided, styled placeholder if not
  let imageBlock = '';
  if (currentImageUrl) {
    imageBlock = `
`;
  } else {
    imageBlock = `

      
      
No Image Provided

      Paste an image URL on the card before generating to see it here
    
`;
  }

  // Replace the image placeholder div
  document.querySelector('.preview-img-placeholder') && (document.querySelector('.preview-wrap').firstElementChild.outerHTML = imageBlock);

  let html = `
${title}
`;
  html += `
${d.intro}
`;

  d.sections.forEach(s => {
    html += `
${s.h2}
`;
    const paras = s.body.split(/\n\n+/);
    paras.forEach(p => {
      p = p.trim();
      if (!p) return;
      if (p.toLowerCase().startsWith('
    ') || p.toLowerCase().startsWith('
      ')) {
        html += p;
      } else {
        html += `
      ${p}
      `;
      }
    });
  });

  html += `
    
      
      
      ${d.cta_heading}
      
      
      ${d.cta_body}
      
      Request a Quote
    
      `;

  // Rebuild the preview wrap cleanly
  document.querySelector('.preview-wrap').innerHTML = `
    ${imageBlock}
    
      
      
      SEO Meta Description (copy into Rank Math)
      
      
      ${d.metaDesc}
      
    
      
    
      ${html}
      `;
}

// ΓÏÉΓÏÉΓÏÉΓÏÉΓÏÉΓÏÉΓÏÉΓÏÉΓÏÉΓÏÉΓÏÉΓÏÉΓÏÉΓÏÉΓÏÉΓÏÉΓÏÉΓÏÉΓÏÉΓÏÉΓÏÉΓÏÉΓÏÉΓÏÉΓÏÉΓÏÉΓÏÉΓÏÉΓÏÉΓÏÉΓÏÉΓÏÉΓÏÉΓÏÉΓÏÉΓÏÉΓÏÉΓÏÉΓÏÉΓÏÉΓÏÉΓÏÉΓÏÉΓÏÉ
function copyCode() {
  const code = document.getElementById('code-content').textContent;
  navigator.clipboard.writeText(code).then(() => {
    const btn = document.getElementById('copy-btn');
    btn.textContent = 'Copied!';
    btn.classList.add('copied');
    setTimeout(() => {
      btn.textContent = 'Copy All Code';
      btn.classList.remove('copied');
    }, 2500);
  }).catch(() => {
    const ta = document.createElement('textarea');
    ta.value = code;
    document.body.appendChild(ta);
    ta.select();
    document.execCommand('copy');
    document.body.removeChild(ta);
    const btn = document.getElementById('copy-btn');
    btn.textContent = 'Copied!';
    btn.classList.add('copied');
    setTimeout(() => {
      btn.textContent = 'Copy All Code';
      btn.classList.remove('copied');
    }, 2500);
  });
}
